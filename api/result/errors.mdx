## AppError

Discriminated union of all possible application errors. Use this as the `E` type in `Result<T, AppError>`.

```typescript
type AppError =
  | ValidationError
  | NotFoundError
  | UnauthorizedError
  | ForbiddenError
  | ConflictError
  | BusinessError
  | DatabaseError
  | ExternalServiceError
  | InternalServerError
  | TooManyRequestsError
```

## BaseAppError

Base structure shared by all application errors.

```typescript
interface BaseAppError {
  readonly type: string;
  readonly message: string;
  readonly statusCode: number;
  readonly timestamp: string;
  readonly name?: string;
  readonly stack?: string;
  readonly component?: string;
  readonly metadata?: Record<string, unknown>;
}
```

<ResponseField name="type" type="string" required>
  The error type identifier (e.g., `"VALIDATION_ERROR"`, `"NOT_FOUND"`)
</ResponseField>

<ResponseField name="message" type="string" required>
  Human-readable error message
</ResponseField>

<ResponseField name="statusCode" type="number" required>
  HTTP status code associated with this error
</ResponseField>

<ResponseField name="timestamp" type="string" required>
  ISO 8601 timestamp when the error occurred
</ResponseField>

<ResponseField name="name" type="string">
  Error class name (e.g., `"ValidationError"`, `"NotFoundError"`)
</ResponseField>

<ResponseField name="stack" type="string">
  Stack trace filtered to show user code location
</ResponseField>

<ResponseField name="component" type="string">
  The component or module where the error originated
</ResponseField>

<ResponseField name="metadata" type="Record<string, unknown>">
  Additional contextual data
</ResponseField>

## Error types

### ValidationError

HTTP 422 - Input data failed validation.

```typescript
type ValidationError = BaseAppError & {
  readonly type: 'VALIDATION_ERROR';
  readonly errors: ValidationFieldError[];
}

interface ValidationFieldError {
  readonly field: string;
  readonly message: string;
  readonly rule?: string;
  readonly value?: unknown;
}
```

<ResponseField name="errors" type="ValidationFieldError[]" required>
  Array of field-level validation errors
</ResponseField>

### NotFoundError

HTTP 404 - Requested resource does not exist.

```typescript
type NotFoundError = BaseAppError & {
  readonly type: 'NOT_FOUND';
  readonly resource?: string;
  readonly resourceId?: string | number;
}
```

<ResponseField name="resource" type="string">
  The type of resource that was not found (e.g., `"User"`, `"Product"`)
</ResponseField>

<ResponseField name="resourceId" type="string | number">
  The ID of the resource that was not found
</ResponseField>

### UnauthorizedError

HTTP 401 - Request is not authenticated.

```typescript
type UnauthorizedError = BaseAppError & {
  readonly type: 'UNAUTHORIZED';
  readonly reason?:
    | 'invalid_token'
    | 'expired_token'
    | 'missing_token'
    | 'invalid_credentials'
    | 'missing_credentials';
}
```

<ResponseField name="reason" type="string">
  Specific reason for unauthorized access: `invalid_token`, `expired_token`, `missing_token`, `invalid_credentials`, or `missing_credentials`
</ResponseField>

### ForbiddenError

HTTP 403 - Authenticated but not allowed.

```typescript
type ForbiddenError = BaseAppError & {
  readonly type: 'FORBIDDEN';
  readonly requiredPermission?: string;
}
```

<ResponseField name="requiredPermission" type="string">
  The permission required to access the resource
</ResponseField>

### ConflictError

```typescript
type ConflictError = BaseAppError & {
  readonly type: 'CONFLICT';

```

<ResponseField name="conflictingField" type="string">
  The field that has a conflicting value
</ResponseField>

<ResponseField name="existingValue" type="unknown">
  The existing value that caused the conflict
</ResponseField>

### BusinessError

HTTP 400 - Business rule violation.

```typescript
type BusinessError = BaseAppError & {
  readonly type: 'BUSINESS_ERROR';

```

<ResponseField name="code" type="string">

### DatabaseError

HTTP 500 - Database operation failed.

```typescript
type DatabaseError = BaseAppError & {
  readonly type: 'DATABASE_ERROR';
  readonly operation?: 'SELECT' | 'INSERT' | 'UPDATE' | 'DELETE' | 'TRANSACTION';
  readonly table?: string;
  readonly cause?: unknown;
}
```

<ResponseField name="operation" type="string">
  The database operation that failed: `SELECT`, `INSERT`, `UPDATE`, `DELETE`, or `TRANSACTION`
</ResponseField>

<ResponseField name="table" type="string">
  The database table involved in the failed operation
</ResponseField>

<ResponseField name="cause" type="unknown">
  The underlying error that caused the database failure
</ResponseField>


## ErrorFactory

The only way to create AppErrors in NduloJS. All methods return typed errors ready to be wrapped with `Err()`.

### validation

Creates a validation error (HTTP 422).

```typescript
validation(
  message: string,
  errors?: ValidationFieldError[],
  component?: string
): ValidationError
```

<ParamField path="message" type="string" required>
  Overall validation error message
</ParamField>

<ParamField path="errors" type="ValidationFieldError[]" default="[]">
  Array of field-level validation errors
</ParamField>

<ParamField path="component" type="string">
  Component where validation failed
</ParamField>

<CodeGroup>

```typescript Basic validation error
return Err(ErrorFactory.validation('Invalid input'));
```

```typescript With field errors
return Err(
  ErrorFactory.validation('Validation failed', [
    { field: 'email', message: 'Must be a valid email' },
    { field: 'age', message: 'Must be at least 18', rule: 'min', value: 16 }
  ])
);
```

</CodeGroup>
